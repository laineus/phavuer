# Setup with CDN

Phavuer requires Phaser 3 and Vue 3.  
Ensure Phavuer is loaded after these libraries.

```html
<script src="https://unpkg.com/phaser@3.90.0/dist/phaser.min.js"></script>
<script src="https://unpkg.com/vue@3.5.24/dist/vue.global.prod.js"></script>
<script src="https://unpkg.com/phavuer/dist/phavuer.min.js"></script>
```

```html
<div id="app"></div>
```

```js
const { Game, Scene, Rectangle } = Phavuer
const MainScene = {
  components: { Game, Scene, Rectangle },
  template: `<Game>
              <Scene name="MainScene">
                <Rectangle :x="100" :y="100" :width="100" :height="100" :fillColor="0xAAAAAA" />
              </Scene>
            </Game>`,
  setup () {
    return {}
  }
}
document.addEventListener('DOMContentLoaded', () => {
  const app = Vue.createApp(MainScene)
  app.mount('#app')
})
```
